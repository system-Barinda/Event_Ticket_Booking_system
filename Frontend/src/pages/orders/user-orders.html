No events found.

"; return; } container.innerHTML = ""; data.events.forEach(event => { container.innerHTML += `

${event.title}

Date: ${event.date}

Location: ${event.location}

Price: $${event.price}

Available: ${event.available_tickets}

`; }); } loadEvents();
